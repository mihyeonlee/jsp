조건이 있어야함.

SELECT ROWNUM, a.*
FROM
    (SELECT * 
     FROM users
     ORDER BY userid) a
WHERE ROWNUM BETWEEN 1 AND 7;

ROWNUM은 1부터 있는 형태만 읽어올 수 있다. 
ex) WHERE ROWNUM BETWEEN 1 AND 7; ==> OK
	WHERE ROWNUM BETWEEN 8 AND 14; ==> 결과 없음
	
--> 때문에 이렇게 작성해야함
SELECT *
FROM
    (SELECT ROWNUM rn, a.*
    FROM
        (SELECT * 
         FROM users
         ORDER BY userid) a)
WHERE rn BETWEEN #{page} AND #{pageSize}
WHERE rn BETWEEN 1 AND 7
WHERE rn BETWEEN 8 AND 14
WHERE rn BETWEEN 15 AND 21;

SELECT *

int page = 2, int pageSize = 7

SELECT *
FROM
    (SELECT ROWNUM rn, a.*
    FROM
        (SELECT * 
         FROM users
         ORDER BY userid) a)
WHERE rn BETWEEN (:page-1)*:pageSize+1 AND :page * :pageSize;
